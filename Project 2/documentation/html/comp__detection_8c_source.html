270 unsigned short in_cksum(unsigned short *addr, int len)
271 {
272  register int sum = 0;
273  u_short answer = 0;
274  register u_short *w = addr;
275  register int nleft = len;
276 /*
277 * Our algorithm is simple, using a 32 bit accumulator (sum), we add
278 * sequential 16 bit words to it, and at the end, fold back all the
279 * carry bits from the top 16 bits into the lower 16 bits.
280 */
281  while (nleft > 1)
282  {
283  sum += *w++;
284  nleft -= 2;
285  }
286 /* mop up an odd byte, if necessary */
287  if (nleft == 1)
288  {
289  *(u_char *) (&answer) = *(u_char *) w;
290  sum += answer;
291  }
292 /* add back carry outs from top 16 bits to low 16 bits */
293  sum = (sum >> 16) + (sum & 0xffff); /* add hi 16 to low 16 */
294  sum += (sum >> 16); /* add carry */
295  answer = ~sum; /* truncate to 16 bits */
296  return (answer);
297 }
